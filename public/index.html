ESP32 Connection Setup

How to connect your ESP32

  1. Connect a DHT22 sensor to pin 4, and a GPS module to pins 16 (RX) & 17 (TX).
  2. Make sure your ESP32 and this server are on the same WiFi network.
  3. Replace YOUR_WIFI_SSID, YOUR_WIFI_PASSWORD, and YOUR_SERVER_IP in the code below.
  4. Upload the code via Arduino IDE and the sensor data will appear in the web app automatically!

ESP32 Firmware Code (Arduino C++)

#include <WiFi.h>
#include <HTTPClient.h>
#include <DHT.h>
#include <TinyGPSPlus.h>
#include <HardwareSerial.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
// Replace YOUR_SERVER_IP with your computer's local IP (e.g. 192.168.1.15)
const char* serverURL = "http://YOUR_SERVER_IP:3000/api/sensor";

#define DHTPIN 4
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);
HardwareSerial GPSSerial(1);
TinyGPSPlus gps;

void setup() {
  Serial.begin(115200);
  dht.begin();
  GPSSerial.begin(9600, SERIAL_8N1, 16, 17);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
}

void loop() {
  while (GPSSerial.available() > 0) gps.encode(GPSSerial.read());

  if (WiFi.status() == WL_CONNECTED) {
    float temp = dht.readTemperature();
    float hum = dht.readHumidity();
    float heatIndex = dht.computeHeatIndex(temp, hum, false);
    float lat = gps.location.isValid() ? gps.location.lat() : 28.6148;
    float lng = gps.location.isValid() ? gps.location.lng() : 77.2082;

    String json = "{\"id\":\"HI-IOT-101\",\"temp\":" + String(temp) + ",\"humidity\":" + String(hum) + ",\"heat_index\":" + String(heatIndex) + ",\"lat\":" + String(lat, 6) + ",\"lng\":" + String(lng, 6) + ",\"battery\":85}";

    HTTPClient http;
    http.begin(serverURL);
    http.addHeader("Content-Type", "application/json");
    http.POST(json);
    http.end();
  }
  delay(300000); // 5 minutes
}
